```bash
docker run -d \
  --name=freshrss \
  -e PUID=$(id -u) \
  -e PGID=$(id -u) \
  -e TZ=Europe/Paris \
  -p 80:80 \
  -v /DATA/AppData/FreshRSS/config:/config \
  --restart unless-stopped \
  lscr.io/linuxserver/freshrss
```
